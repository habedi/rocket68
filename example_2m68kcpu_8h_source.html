1242/* Indexed addressing modes are encoded as follows:
1243 *
1244 * Base instruction format:
1245 * F E D C B A 9 8 7 6 | 5 4 3 | 2 1 0
1246 * x x x x x x x x x x | 1 1 0 | BASE REGISTER (An)
1247 *
1248 * Base instruction format for destination EA in move instructions:
1249 * F E D C | B A 9 | 8 7 6 | 5 4 3 2 1 0
1250 * x x x x | BASE REG | 1 1 0 | X X X X X X (An)
1251 *
1252 * Brief extension format:
1253 * F | E D C | B | A 9 | 8 | 7 6 5 4 3 2 1 0
1254 * D/A | REGISTER | W/L | SCALE | 0 | DISPLACEMENT
1255 *
1256 * Full extension format:
1257 * F E D C B A 9 8 7 6 5 4 3 2 1 0
1258 * D/A | REGISTER | W/L | SCALE | 1 | BS | IS | BD SIZE | 0 | I/IS
1259 * BASE DISPLACEMENT (0, 16, 32 bit) (bd)
1260 * OUTER DISPLACEMENT (0, 16, 32 bit) (od)
1261 *
1262 * D/A: 0 = Dn, 1 = An (Xn)
1263 * W/L: 0 = W (sign extend), 1 = L (.SIZE)
1264 * SCALE: 00=1, 01=2, 10=4, 11=8 (*SCALE)
1265 * BS: 0=add base reg, 1=suppress base reg (An suppressed)
1266 * IS: 0=add index, 1=suppress index (Xn suppressed)
1267 * BD SIZE: 00=reserved, 01=NULL, 10=Word, 11=Long (size of bd)
1268 *
1269 * IS I/IS Operation
1270 * 0 000 No Memory Indirect
1271 * 0 001 indir prex with null outer
1272 * 0 010 indir prex with word outer
1273 * 0 011 indir prex with long outer
1274 * 0 100 reserved
1275 * 0 101 indir postx with null outer
1276 * 0 110 indir postx with word outer
1277 * 0 111 indir postx with long outer
1278 * 1 000 no memory indirect
1279 * 1 001 mem indir with null outer
1280 * 1 010 mem indir with word outer
1281 * 1 011 mem indir with long outer
1282 * 1 100-111 reserved
1283 */
1284static inline uint m68ki_get_ea_ix(uint An)
1285{
1286 /* An = base register */
1287 uint extension = m68ki_read_imm_16();
1288 uint Xn = 0; /* Index register */
1289 uint bd = 0; /* Base Displacement */
1290 uint od = 0; /* Outer Displacement */
1291
1292 if(CPU_TYPE_IS_010_LESS(CPU_TYPE))
1293 {
1294 /* Calculate index */
1295 Xn = REG_DA[extension>>12]; /* Xn */
1296 if(!BIT_B(extension)) /* W/L */
1297 Xn = MAKE_INT_16(Xn);
1298
1299 /* Add base register and displacement and return */
1300 return An + Xn + MAKE_INT_8(extension);
1301 }
1302
1303 /* Brief extension format */
1304 if(!BIT_8(extension))
1305 {
1306 /* Calculate index */
1307 Xn = REG_DA[extension>>12]; /* Xn */
1308 if(!BIT_B(extension)) /* W/L */
1309 Xn = MAKE_INT_16(Xn);
1310 /* Add scale if proper CPU type */
1311 if(CPU_TYPE_IS_EC020_PLUS(CPU_TYPE))
1312 Xn <<= (extension>>9) & 3; /* SCALE */
1313
1314 /* Add base register and displacement and return */
1315 return An + Xn + MAKE_INT_8(extension);
1316 }
1317
1318 /* Full extension format */
1319
1320 USE_CYCLES(m68ki_ea_idx_cycle_table[extension&0x3f]);
1321
1322 /* Check if base register is present */
1323 if(BIT_7(extension)) /* BS */
1324 An = 0; /* An */
1325
1326 /* Check if index is present */
1327 if(!BIT_6(extension)) /* IS */
1328 {
1329 Xn = REG_DA[extension>>12]; /* Xn */
1330 if(!BIT_B(extension)) /* W/L */
1331 Xn = MAKE_INT_16(Xn);
1332 Xn <<= (extension>>9) & 3; /* SCALE */
1333 }
1334
1335 /* Check if base displacement is present */
1336 if(BIT_5(extension)) /* BD SIZE */
1337 bd = BIT_4(extension) ? m68ki_read_imm_32() : (uint32)MAKE_INT_16(m68ki_read_imm_16());
1338
1339 /* If no indirect action, we are done */
1340 if(!(extension&7)) /* No Memory Indirect */
1341 return An + bd + Xn;
1342
1343 /* Check if outer displacement is present */
1344 if(BIT_1(extension)) /* I/IS: od */
1345 od = BIT_0(extension) ? m68ki_read_imm_32() : (uint32)MAKE_INT_16(m68ki_read_imm_16());
1346
1347 /* Postindex */
1348 if(BIT_2(extension)) /* I/IS: 0 = preindex, 1 = postindex */
1349 return m68ki_read_32(An + bd) + Xn + od;
1350
1351 /* Preindex */
1352 return m68ki_read_32(An + bd + Xn) + od;
1353}

2064/* Service an interrupt request and start exception processing */
2065static inline void m68ki_exception_interrupt(uint int_level)
2066{
2067 uint vector;
2068 uint sr;
2069 uint new_pc;
2070
2071 #if M68K_EMULATE_ADDRESS_ERROR == OPT_ON
2072 if(CPU_TYPE_IS_000(CPU_TYPE))
2073 {
2074 CPU_INSTR_MODE = INSTRUCTION_NO;
2075 }
2076 #endif /* M68K_EMULATE_ADDRESS_ERROR */
2077
2078 /* Turn off the stopped state */
2079 CPU_STOPPED &= ~STOP_LEVEL_STOP;
2080
2081 /* If we are halted, don't do anything */
2082 if(CPU_STOPPED)
2083 return;
2084
2085 /* Acknowledge the interrupt */
2086 vector = m68ki_int_ack(int_level);
2087
2088 /* Get the interrupt vector */
2089 if(vector == M68K_INT_ACK_AUTOVECTOR)
2090 /* Use the autovectors. This is the most commonly used implementation */
2091 vector = EXCEPTION_INTERRUPT_AUTOVECTOR+int_level;
2092 else if(vector == M68K_INT_ACK_SPURIOUS)
2093 /* Called if no devices respond to the interrupt acknowledge */
2094 vector = EXCEPTION_SPURIOUS_INTERRUPT;
2095 else if(vector > 255)
2096 {
2097 M68K_DO_LOG_EMU((M68K_LOG_FILEHANDLE "%s at %08x: Interrupt acknowledge returned invalid vector $%x\n",
2098 m68ki_cpu_names[CPU_TYPE], ADDRESS_68K(REG_PC), vector));
2099 return;
2100 }
2101
2102 /* Start exception processing */
2103 sr = m68ki_init_exception();
2104
2105 /* Set the interrupt mask to the level of the one being serviced */
2106 FLAG_INT_MASK = int_level<<8;
2107
2108 /* Get the new PC */
2109 new_pc = m68ki_read_data_32((vector<<2) + REG_VBR);
2110
2111 /* If vector is uninitialized, call the uninitialized interrupt vector */
2112 if(new_pc == 0)
2113 new_pc = m68ki_read_data_32((EXCEPTION_UNINITIALIZED_INTERRUPT<<2) + REG_VBR);
2114
2115 /* Generate a stack frame */
2116 m68ki_stack_frame_0000(REG_PC, sr, vector);
2117 if(FLAG_M && CPU_TYPE_IS_EC020_PLUS(CPU_TYPE))
2118 {
2119 /* Create throwaway frame */
2120 m68ki_set_sm_flag(FLAG_S); /* clear M */
2121 sr |= 0x2000; /* Same as SR in master stack frame except S is forced high */
2122 m68ki_stack_frame_0001(REG_PC, sr, vector);
2123 }
2124
2125 m68ki_jump(new_pc);
2126
2127 /* Defer cycle counting until later */
2128 USE_CYCLES(CYC_EXCEPTION[vector]);
2129
2130#if !M68K_EMULATE_INT_ACK
2131 /* Automatically clear IRQ if we are not using an acknowledge scheme */
2132 CPU_INT_LEVEL = 0;
2133#endif /* M68K_EMULATE_INT_ACK */
2134}
